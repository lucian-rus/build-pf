#include <linux.init.h>
#include <linux/module.h>

MODULE_LICENSE("GPL");

static int <template_module_name>_init(void) {
    printk("\n-----------> module <template_module_name> - hello!\n");
    return 0;
}

static void <template_module_name>_exit(void) {
    printk("\n-----------> module <template_module_name> - goodbye!\n");
}

module_init(<template_module_name>_init);
module_exit(<template_module_name>_exit);
